5. SELECT 
    CASE
        WHEN LEFT(posfou, 2) = '75' THEN '75'
        WHEN LEFT(posfou, 2) = '78' THEN '78'
        WHEN LEFT(posfou, 2) = '92' THEN '92'
        WHEN LEFT(posfou, 2) = '77' THEN '77'
    END AS posfou,
    nomfou
FROM 
    fournis
WHERE 
    LEFT(posfou, 2) IN ('75', '78', '92', '77')
ORDER BY 
    posfou DESC, 
    nomfou ASC;
